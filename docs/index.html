jQuery Timeline v0.0.1

Basic usage


var date_range = [];
var date = new Date();
date_range.push(date);
for (var i = 0; i < 3; i++) {
  date = new Date(date.setUTCDate(date.getUTCDate() + 1));
  date_range.push(date);
}
$("#basic-timeline").timeline({
  dates: date_range, // array of dates
  onSelection: function(date) {
    // event when a date is selected
    console.log("selected: " + date);
  }
});
